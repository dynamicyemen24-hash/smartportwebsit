-- ======================================================
-- 1. تهيئة البيئة والمخططات Organization (Schemas)
-- ======================================================
-- نستخدم مخططات منفصلة لتنظيم قاعدة البيانات بشكل معياري.
CREATE SCHEMA IF NOT EXISTS app_auth; -- لإدارة الصلاحيات والأدوار المرنة
CREATE SCHEMA IF NOT EXISTS app_crm;  -- لكل ما يخص CRM (عملاء، فرص، أنشطة)
CREATE SCHEMA IF NOT EXISTS app_erp;  -- لكل ما يخص ERP (منتجات، مخزون، فواتير، مشتريات)
CREATE SCHEMA IF NOT EXISTS app_audit; -- لجدول مراجعة شامل بدون تداخل مع البيانات التشغيلية
CREATE SCHEMA IF NOT EXISTS app_system; -- للوظائف والعمليات المساعدة مثل الإشعارات والملفات

-- إضافة تعليقات لتوثيق كل مخطط (Best Practice)
COMMENT ON SCHEMA app_auth IS 'يدير صلاحيات المستخدمين، الأدوار (Roles)، والانتماء للمؤسسات.';
COMMENT ON SCHEMA app_crm IS 'يدير جهات الاتصال، العملاء المتوقعين (Leads)، الفرص (Opportunities)، والأنشطة.';
COMMENT ON SCHEMA app_erp IS 'يدير المنتجات، المخزون، أوامر البيع والشراء، والفواتير والمحاسبة.';
COMMENT ON SCHEMA app_audit IS 'يحتفظ بسجل تدقيق كامل (Audit Log) لجميع التغييرات الهامة.';
COMMENT ON SCHEMA app_system IS 'يدير الملفات المرفقة، قوالب البريد الإلكتروني، وسجلات النظام.';


-- ======================================================
-- 2. تعريف أنواع ENUMs للقيم الثابتة (موصى به من Supabase)
-- ======================================================
-- استخدام الـ ENUMs أفضل من جداول البحث الثابتة للأداء والوضوح، مع العلم أن تغييرها يتطلب ترحيل (Migration).
-- Supabase توصي باستخدام الـ ENUMs للقوائم الصغيرة والثابتة التي نادرًا ما تتغير.
CREATE TYPE app_auth.membership_status AS ENUM ('invited', 'active', 'suspended', 'removed');
COMMENT ON TYPE app_auth.membership_status IS 'حالة عضوية المستخدم في المؤسسة.';

CREATE TYPE app_crm.lead_status AS ENUM ('new', 'contacted', 'qualified', 'unqualified', 'converted');
COMMENT ON TYPE app_crm.lead_status IS 'حالة العميل المتوقع في مسار المبيعات.';

CREATE TYPE app_crm.opportunity_stage AS ENUM ('prospecting', 'qualification', 'needs_analysis', 'value_proposition', 'negotiation', 'closed_won', 'closed_lost');
COMMENT ON TYPE app_crm.opportunity_stage IS 'مراحل فرصة البيع.';

CREATE TYPE app_crm.activity_type AS ENUM ('call', 'email', 'meeting', 'task', 'note');
COMMENT ON TYPE app_crm.activity_type IS 'نوع النشاط المسجل.';

CREATE TYPE app_erp.order_status AS ENUM ('draft', 'confirmed', 'shipped', 'delivered', 'cancelled', 'invoiced');
COMMENT ON TYPE app_erp.order_status IS 'حالة أمر البيع.';

CREATE TYPE app_erp.purchase_order_status AS ENUM ('draft', 'sent', 'confirmed', 'received', 'cancelled');
COMMENT ON TYPE app_erp.purchase_order_status IS 'حالة أمر الشراء.';

CREATE TYPE app_erp.invoice_status AS ENUM ('draft', 'sent', 'paid', 'overdue', 'cancelled');
COMMENT ON TYPE app_erp.invoice_status IS 'حالة الفاتورة.';

CREATE TYPE app_erp.payment_type AS ENUM ('cash', 'bank_transfer', 'check', 'credit_card', 'online');
COMMENT ON TYPE app_erp.payment_type IS 'طريقة الدفع.';

CREATE TYPE app_erp.stock_movement_type AS ENUM ('in', 'out', 'adjustment');
COMMENT ON TYPE app_erp.stock_movement_type IS 'نوع حركة المخزون.';


-- ======================================================
-- 3. بناء الجداول الأساسية (Core Tables)
-- ======================================================

-- 3.1 جدول المؤسسات، أساس الـ Multi-Tenancy
CREATE TABLE IF NOT EXISTS app_auth.organizations (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    name text NOT NULL,
    slug text UNIQUE NOT NULL,
    logo_url text,
    tax_id text UNIQUE,
    email text,
    phone text,
    address text,
    city text,
    state text,
    postal_code text,
    country text,
    settings jsonb DEFAULT '{}'::jsonb,
    is_active boolean DEFAULT true,
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_auth.organizations IS 'المؤسسة (الشركة) هي صاحب البيانات وهي أساس العزل البيئي (Tenant Isolation).';

-- 3.2 توسيع جدول المستخدمين الأصلي في Supabase، لربط كل مستخدم بمعلوماته الإضافية
CREATE TABLE IF NOT EXISTS app_auth.profiles (
    id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    first_name text,
    last_name text,
    avatar_url text,
    phone text,
    job_title text,
    timezone text DEFAULT 'UTC',
    settings jsonb DEFAULT '{}'::jsonb,
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_auth.profiles IS 'معلومات إضافية لكل مستخدم، مرتبطة بجدول auth.users الأصلي في Supabase.';

-- 3.3 أدوار المستخدمين (Member Roles) داخل كل مؤسسة.
-- هذا يستخدم لإدارة صلاحيات دقيقة على مستوى التطبيق، وليس دور PostgreSQL.
CREATE TABLE IF NOT EXISTS app_auth.members (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    role text NOT NULL,
    status app_auth.membership_status DEFAULT 'invited'::app_auth.membership_status,
    invited_by uuid REFERENCES auth.users(id),
    joined_at timestamptz,
    left_at timestamptz,
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now(),
    UNIQUE(organization_id, user_id)
);
COMMENT ON TABLE app_auth.members IS 'عضو (مستخدم) مرتبط بمؤسسة معينة، صلب التحكم في الصلاحيات ونماذج الـ RLS.';

-- 3.4 إدارة الأذونات الديناميكية للأدوار (Dynamic Role-Based Permissions)
CREATE TABLE IF NOT EXISTS app_auth.role_permissions (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    role text NOT NULL,
    permission text NOT NULL,
    resource text NOT NULL,
    created_at timestamptz DEFAULT now(),
    UNIQUE(role, permission, resource)
);
COMMENT ON TABLE app_auth.role_permissions IS 'يحدد الأذونات (مثل create, read, update, delete) لكل دور على مورد معين.';


-- ======================================================
-- 4. بناء جداول الـ CRM
-- ======================================================

-- 4.1 جهات الاتصال (Contacts) - يمكن أن تكون فرداً أو جزءاً من شركة
-- استلهمنا نموذج الـ Identity من appmaster.io حيث يتم الفصل بين هوية الشخص والشركة التابعة له.
CREATE TABLE IF NOT EXISTS app_crm.contacts (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    first_name text,
    last_name text,
    email text,
    phone text,
    mobile text,
    job_title text,
    department text,
    is_primary boolean DEFAULT false,
    birthday date,
    notes text,
    owner_id uuid REFERENCES auth.users(id),
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_crm.contacts IS 'جهة اتصال (شخص) مرتبطة بمؤسسة، يمكن أن يكون عميلاً، مورداً، أو موظفاً.';

-- 4.2 الشركات (Companies) للأعمال B2B
CREATE TABLE IF NOT EXISTS app_crm.companies (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    name text NOT NULL,
    website text,
    industry text,
    annual_revenue numeric,
    employees_count integer,
    owner_id uuid REFERENCES auth.users(id),
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_crm.companies IS 'شركة (عميل محتمل أو حالي) في نظام B2B.';

-- 4.3 العملاء المتوقعون (Leads)
CREATE TABLE IF NOT EXISTS app_crm.leads (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    contact_id uuid REFERENCES app_crm.contacts(id),
    company_id uuid REFERENCES app_crm.companies(id),
    source text,
    status app_crm.lead_status DEFAULT 'new'::app_crm.lead_status,
    assigned_to uuid REFERENCES auth.users(id),
    converted_to_opportunity_id uuid, -- سيتم ربطها بالجداول التالية
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_crm.leads IS 'عميل متوقع قبل التأهيل.';

-- 4.4 الفرص (Opportunities)
CREATE TABLE IF NOT EXISTS app_crm.opportunities (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,

    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),

COMMENT ON TABLE app_crm.opportunities IS 'فرصة مبيعات مؤهلة.';

CREATE TABLE IF NOT EXISTS app_crm.activities (

    created_by uuid REFERENCES auth.users(id),

CREATE TABLE IF NOT EXISTS app_erp.product_categories (

COMMENT ON TABLE app_erp.product_categories IS 'تصنيفات المنتجات الهرمية.';

CREATE TABLE IF NOT EXISTS app_erp.products (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    name text NOT NULL,
    sku text UNIQUE,
    description text,
    category_id uuid REFERENCES app_erp.product_categories(id),
    unit_price numeric NOT NULL CHECK (unit_price >= 0),
    cost_price numeric CHECK (cost_price >= 0),
    reorder_level integer DEFAULT 0,
    is_active boolean DEFAULT true,
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()

COMMENT ON TABLE app_erp.products IS 'جدول المنتجات الأساسي.';

-- 5.2 جداول المخزون
CREATE TABLE IF NOT EXISTS app_erp.warehouses (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    name text NOT NULL,
    location text,
    is_active boolean DEFAULT true,
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_erp.warehouses IS 'المستودعات ومواقع التخزين.';

-- سجل حركة المخزون (Stock Movements)
CREATE TABLE IF NOT EXISTS app_erp.stock_movements (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    product_id uuid NOT NULL REFERENCES app_erp.products(id),
    warehouse_id uuid NOT NULL REFERENCES app_erp.warehouses(id),
    movement_type app_erp.stock_movement_type NOT NULL,
    quantity integer NOT NULL,
    reference_type text, -- مثل 'sales_order', 'purchase_order'
    reference_id uuid,
    created_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_erp.stock_movements IS 'سجل حركات الدخول والخروج من المخزون. هذا الجدول أساسي للمراجعة.';

-- رصيد المخزون الحالي: يمكن حسابه عن طريق تجميع stock_movements. للحصول على أفضل أداء، يمكننا استخدام View أو تحديث جدول ملخص.
-- سنقوم بإنشاء View بسيط للرصيد الحالي:
CREATE OR REPLACE VIEW app_erp.current_stock AS
SELECT
    organization_id,
    product_id,
    warehouse_id,
    SUM(CASE
        WHEN movement_type = 'in' THEN quantity
        WHEN movement_type = 'out' THEN -quantity
        ELSE quantity -- للـ adjustment، يمكن معاملتها كإضافة أو خصم بناءً على القيمة
    END) AS current_quantity
FROM app_erp.stock_movements
GROUP BY organization_id, product_id, warehouse_id;

-- 5.3 أوامر البيع (Sales Orders)
CREATE TABLE IF NOT EXISTS app_erp.sales_orders (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    order_number text UNIQUE,
    contact_id uuid REFERENCES app_crm.contacts(id),
    company_id uuid REFERENCES app_crm.companies(id),
    order_date date DEFAULT CURRENT_DATE,
    expected_date date,
    status app_erp.order_status DEFAULT 'draft'::app_erp.order_status,
    notes text,
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_erp.sales_orders IS 'أوامر البيع الصادرة للعملاء.';

CREATE TABLE IF NOT EXISTS app_erp.sales_order_lines (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    sales_order_id uuid NOT NULL REFERENCES app_erp.sales_orders(id) ON DELETE CASCADE,
    product_id uuid NOT NULL REFERENCES app_erp.products(id),
    quantity integer NOT NULL CHECK (quantity > 0),
    unit_price numeric NOT NULL,
    discount_amount numeric DEFAULT 0,
    total_amount numeric GENERATED ALWAYS AS ((quantity * unit_price) - discount_amount) STORED,
    created_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_erp.sales_order_lines IS 'تفاصيل منتجات أمر البيع.';

-- 5.4 الفواتير والمدفوعات

    contact_id uuid REFERENCES app_crm.contacts(id),

    status app_erp.invoice_status DEFAULT 'draft'::app_erp.invoice_status,
    created_by uuid REFERENCES auth.users(id),
    updated_by uuid REFERENCES auth.users(id),
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_erp.invoices IS 'الفواتير الصادرة والواردة.';

CREATE TABLE IF NOT EXISTS app_erp.payments (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES app_auth.organizations(id) ON DELETE CASCADE,
    invoice_id uuid NOT NULL REFERENCES app_erp.invoices(id),
    amount numeric NOT NULL,
    payment_date date DEFAULT CURRENT_DATE,
    payment_type app_erp.payment_type,
    reference_number text,
    notes text,
    created_by uuid REFERENCES auth.users(id),

-- ======================================================
-- 6. نظام مراجعة شامل (Comprehensive Audit System)
-- ======================================================
-- جدول مراجعة واحد لكل التغييرات الهامة، لتتبع الأمان والامتثال.
-- يمكن تمديد هذا النظام لاحقاً بتشغيله مباشرة على الجداول.
CREATE TABLE IF NOT EXISTS app_audit.audit_logs (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid REFERENCES app_auth.organizations(id) ON DELETE SET NULL,
    table_name text NOT NULL,
    record_id uuid NOT NULL,
    action text NOT NULL, -- INSERT, UPDATE, DELETE
    old_data jsonb,
    new_data jsonb,
    changed_by uuid REFERENCES auth.users(id),
    changed_at timestamptz DEFAULT now()
);
COMMENT ON TABLE app_audit.audit_logs IS 'سجل تدقيق لكل التغييرات الهامة في قاعدة البيانات.';


-- ======================================================
-- 7. الوظائف والإجراءات المخزنة (Functions & Procedures)
-- ======================================================

-- 7.1 دالة لتحديث updated_at تلقائياً (لأي جدول يحتوي على الأعمدة created_at/updated_at)
-- هذه أداة ذكية لتوفير الوقت والجهد. سنقوم باستخدامها مع Trigger لاحقاً.
CREATE OR REPLACE FUNCTION app_system.update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 7.2 دالة لتسجيل التغييرات في جدول audit_logs تلقائياً.
-- يمكن تطبيقها عبر Trigger على الجداول الحساسة.
CREATE OR REPLACE FUNCTION app_audit.if_modified_func()
RETURNS TRIGGER AS $$
BEGIN
    IF (TG_OP = 'INSERT') THEN
        INSERT INTO app_audit.audit_logs (organization_id, table_name, record_id, action, new_data, changed_by)
        VALUES (
            (SELECT organization_id FROM (
                SELECT organization_id FROM app_crm.contacts WHERE id = NEW.id UNION ALL
                SELECT organization_id FROM app_crm.companies WHERE id = NEW.id UNION ALL
                SELECT organization_id FROM app_erp.products WHERE id = NEW.id
            ) AS org),
            TG_TABLE_NAME,
            NEW.id,
            'INSERT',
            to_jsonb(NEW),
            auth.uid()
        );
        RETURN NEW;
    ELSIF (TG_OP = 'UPDATE') THEN
        INSERT INTO app_audit.audit_logs (organization_id, table_name, record_id, action, old_data, new_data, changed_by)
        VALUES (
            (SELECT organization_id FROM (
                SELECT organization_id FROM app_crm.contacts WHERE id = OLD.id UNION ALL
                SELECT organization_id FROM app_crm.companies WHERE id = OLD.id UNION ALL
                SELECT organization_id FROM app_erp.products WHERE id = OLD.id
            ) AS org),
            TG_TABLE_NAME,
            OLD.id,
            'UPDATE',
            to_jsonb(OLD),
            to_jsonb(NEW),
            auth.uid()
        );
        RETURN NEW;
    ELSIF (TG_OP = 'DELETE') THEN
        INSERT INTO app_audit.audit_logs (organization_id, table_name, record_id, action, old_data, changed_by)
        VALUES (
            (SELECT organization_id FROM (
                SELECT organization_id FROM app_crm.contacts WHERE id = OLD.id UNION ALL
                SELECT organization_id FROM app_crm.companies WHERE id = OLD.id UNION ALL
                SELECT organization_id FROM app_erp.products WHERE id = OLD.id
            ) AS org),
            TG_TABLE_NAME,
            OLD.id,
            'DELETE',
            to_jsonb(OLD),
            auth.uid()
        );
        RETURN OLD;
    END IF;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 8.2 سياسة الوصول الأساسية: لا يمكن رؤية أو تعديل إلا البيانات المرتبطة بمؤسسة المستخدم.
-- هذه السياسة العامة، يمكن تطويرها لمنح صلاحيات أدق (مثل مدير يمكنه رؤية كل شيء بالكامل).
-- المبدأ: المستخدم يُرى في جدول members ضمن مؤسسة معينة، ويصل فقط لبيانات تلك المؤسسة.

-- الدالة المساعدة لتحديد معرف المؤسسة للمستخدم الحالي.
-- افتراضاً أن كل مستخدم ينتمي لمؤسسة واحدة فقط. للتعددية يمكن تمديدها.
CREATE OR REPLACE FUNCTION app_auth.get_current_organization_id()
RETURNS uuid AS $$
    SELECT organization_id FROM app_auth.members WHERE user_id = auth.uid() AND status = 'active' LIMIT 1;
$$ LANGUAGE sql STABLE SECURITY DEFINER;

-- مثال على سياسة RLS لجدول contacts:
-- يقرأ فقط جهات الاتصال التي تنتمي لمؤسسة المستخدم.
CREATE POLICY "Users can view contacts of their organization" ON app_crm.contacts
    FOR SELECT USING (organization_id = app_auth.get_current_organization_id());

-- سياسة التعديل: يمكن للمستخدم تعديل جهات الاتصال في مؤسسته فقط.
CREATE POLICY "Users can modify contacts of their organization" ON app_crm.contacts
    FOR ALL USING (organization_id = app_auth.get_current_organization_id());

-- يجب تكرار نفس المبدأ على الجداول الأخرى.
-- على سبيل المثال:
CREATE POLICY "Users can view products of their organization" ON app_erp.products
    FOR SELECT USING (organization_id = app_auth.get_current_organization_id());

CREATE POLICY "Users can modify products of their organization" ON app_erp.products
    FOR ALL USING (organization_id = app_auth.get_current_organization_id());
